def sag(f,amp1,amp2):
    sample = 3000
    f = 20.0
    t1 = 1/f
    t2 = 1/f
    w = 2.0 * np.pi * f
    amp1 = 250
    amp2 = 100
    print(sample,f,t1,'%.2f' %w,amp1)
    x = np.linspace(1,t1,sample)
    y1 = amp1 * np.sin(x[0:1000]*w)
    y2 = amp2* np.sin(x[1000:2000]*w)
    y3 = amp1 * np.sin(x[2000:3000]*w)
    y = np.append(y2,y1)
    y = np.append(y3,y)
    plt.style.use('dark_background')
    plt.subplot(111)
    plt.plot(x[::3],y[::3],'-y')
    #plt.plot(x,np.zeros(len(x)))
    plt.grid(True)
    plt.xlabel('Time (S)')
    plt.ylabel('Voltage (V)')
    plt.title('sin 250V / 50 Hz_Voltage Sag')
    return
